### 수찾기

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int N, A[100'001];

int check(const int target) { // 지금은 bool보다 빨라
	int st=0, en=N-1, mid; // 1e5 안돼
	while (st <= en) {
		mid = st/2 + en/2 + (1&st&en);
		if (A[mid] == target) return 1;
        
		if (A[mid] < target) st = mid + 1;
		else en = mid - 1;
	}
	return 0; // en < st일 경우
}

int main() {
	ios::sync_with_stdio(0); cin.tie(0);
    
	cin >> N;
	for (int i = 0; i < N; ++i) cin >> A[i];
	sort(A, A + N);

	int M, B; cin >> M;
	while (M--) {
		cin >> B;
		cout << check(B) << "\n";
	}
	return 0;
}
```
