```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int N, A[500'000];

int lowerIdx(int target) {
	int st, en, mid; st = 0; en = N-1;
	while (st <= en) {
		mid = (st + en) / 2;
		if (target <= A[mid]) en = mid - 1;
		else st = mid + 1;
	}
	return st;
}

int upperIdx(int target) {
	int st, en, mid; st = 0; en = N-1;
	while (st <= en) {
		mid = (st + en) / 2;
		if (A[mid] <= target) st = mid + 1;
		else en = mid - 1;
	}
	return st;
}

int main() {
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> N;
	for (int i = 0; i < N; ++i) cin >> A[i];
	sort(A, A + N);

	int M, target; cin >> M;
	for (int i = 0; i < M; ++i) {
		cin >> target;
		//cout << upper_bound(&A[0], &A[N], target)-lower_bound(&A[0], &A[N], target) << " "; // 280
		//cout << upper_bound(A, A+N, target)-lower_bound(A, A + N, target) << " "; // 284
		cout << upperIdx(target) - lowerIdx(target) << " "; // 336
	}
	return 0;
}
```
